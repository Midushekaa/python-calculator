from js import document # Get HTML elements expression_input = document.getElementById("expression") result_display = document.getElementById("result") # Button click function def click(event): text = event.target.innerText if text == "=": expr = expression_input.value try: result = str(eval(expr)) except: result = "Error" result_display.innerText = result elif text == "C": expression_input.value = "" result_display.innerText = "" else: expression_input.value += text # Add event listeners to buttons buttons = document.querySelectorAll("button") for btn in buttons: btn.addEventListener("click", click)
